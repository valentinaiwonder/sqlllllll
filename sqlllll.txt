SELECT * FROM PACIENTE 
WHERE IDADE BETWEEN 2 AND 5

SELECT * FROM AGENDAMENTO 
WHERE DATA_HORA BETWEEN '2024-01-01' AND '2024-03-31'

SELECT * FROM MEDICAMENTOS 
WHERE PRECO > 25

SELECT * 
	FROM RECEITA  
   WHERE DATA_EMISSAO  > '2024-06-01'
  	 AND ID_AGENDAMENTO = (SELECT ID_AGENDAMENTO
  	 							FROM AGENDAMENTO 
  								 WHERE id_veterinario = (SELECT id_veterinario
  														   FROM VETERINARIO 
  														  WHERE nome = 'Dr. Ana Souza'))

WHERE PACIENTE.ID_TUTOR  = (SELECT	TUTOR.ID_TUTOR FROM TUTOR WHERE TUTOR.NOME = 'Maria Silva')


SELECT * 
	FROM ITENSRECEITA 
   WHERE ID_ITEM_RECEITA  = (SELECT ID_MEDICAMENTO 
  								 FROM medicamentos WHERE nome = 'Anti-inflamatório Y')

                      		FROM veterinario
